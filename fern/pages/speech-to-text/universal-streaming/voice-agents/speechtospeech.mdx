AssemblyAI's Speech-to-Speech (S2S) API is a native real-time voice conversation endpoint. Unlike a traditional STT → LLM → TTS pipeline, the S2S API handles everything in a single WebSocket connection: it listens to the user, understands what they said, generates a response, and speaks it back — all with low latency.

**Key capabilities:**

- **Built-in VAD** — server-side voice activity detection, no client configuration needed
- **Automatic turn detection** — the model decides when the user has finished speaking
- **Native audio output** — streams PCM16 audio back directly, no separate TTS step
- **Function calling** — register tools and handle `tool.call` / `tool.result` events
- **Barge-in / interruption** — users can interrupt the agent mid-response

<Tabs>
<Tab title="Python">
```python
import asyncio
import base64
import json
import urllib.request
import urllib.parse

import sounddevice as sd
import numpy as np
import websockets

API_KEY = "YOUR_API_KEY"
URL = "wss://speech-to-speech.us.assemblyai.com/v1/realtime"

SAMPLE_RATE = 24000
CHANNELS = 1
DTYPE = "int16"

TOOLS = [
    {
        "type": "function",
        "function": {
            "name": "get_weather",
            "description": "Get the current weather for a city.",
            "parameters": {
                "type": "object",
                "properties": {
                    "location": {
                        "type": "string",
                        "description": "City name, e.g. London"
                    }
                },
                "required": ["location"]
            }
        }
    }
]

def get_weather(location: str) -> dict:
    url = f"https://wttr.in/{urllib.parse.quote(location)}?format=j1"
    with urllib.request.urlopen(url, timeout=5) as resp:
        data = json.loads(resp.read())
    current = data["current_condition"][0]
    return {
        "location": location,
        "temp_c": current["temp_C"],
        "temp_f": current["temp_F"],
        "description": current["weatherDesc"][0]["value"],
    }

async def handle_tool_call(ws, call_id: str, name: str, arguments: str) -> None:
    args = json.loads(arguments)
    try:
        if name == "get_weather":
            result = get_weather(args["location"])
        else:
            result = {"error": f"Unknown tool: {name}"}
    except Exception as e:
        result = {"error": str(e)}

    await ws.send(json.dumps({
        "type": "tool.result",
        "call_id": call_id,
        "result": json.dumps(result),
    }))

async def main():
    headers = {"Authorization": f"Bearer {API_KEY}"}
    async with websockets.connect(URL, additional_headers=headers) as ws:

        # Send session config immediately on connect — before session.ready
        await ws.send(json.dumps({
            "type": "session.update",
            "session": {
                "system_prompt": "You are a helpful assistant. Use get_weather when asked about weather.",
                "greeting": "Hey! Ask me anything — I can even check the weather for you.",
                "tools": TOOLS,
            }
        }))

        loop = asyncio.get_running_loop()
        mic_queue = asyncio.Queue()
        session_ready = asyncio.Event()

        def mic_callback(indata, *_):
            # Gate audio behind session_ready — sending input.audio before
            # session.ready will cause session.error with invalid_format
            if session_ready.is_set():
                loop.call_soon_threadsafe(mic_queue.put_nowait, bytes(indata))

        async def send_audio():
            while True:
                chunk = await mic_queue.get()
                await ws.send(json.dumps({
                    "type": "input.audio",
                    "audio": base64.b64encode(chunk).decode()
                }))

        asyncio.create_task(send_audio())

        with sd.InputStream(samplerate=SAMPLE_RATE, channels=CHANNELS,
                            dtype=DTYPE, callback=mic_callback), \
             sd.OutputStream(samplerate=SAMPLE_RATE, channels=CHANNELS,
                             dtype=DTYPE) as speaker:

            async for message in ws:
                event = json.loads(message)
                t = event.get("type")

                if t == "session.ready":
                    print(f"Ready — start speaking  (session_id={event.get('session_id', '')})")
                    session_ready.set()

                elif t == "input.speech.started":
                    print("Listening...")

                elif t == "input.speech.stopped":
                    print("Processing...")

                elif t == "transcript.user":
                    print(f"You: {event['text']}")

                elif t == "transcript.agent":
                    print(f"Agent: {event['text']}")

                elif t == "tool.call":
                    await handle_tool_call(
                        ws,
                        call_id=event["call_id"],
                        name=event["name"],
                        arguments=event.get("arguments", "{}"),
                    )

                elif t == "reply.audio":
                    pcm = np.frombuffer(base64.b64decode(event["data"]), dtype=np.int16)
                    speaker.write(pcm)

                elif t == "session.error":
                    print(f"Error: {event.get('message')}")
                    break

asyncio.run(main())
```

```javascript
import WebSocket from "ws";
import mic from "mic";
import Speaker from "speaker";

const API_KEY = "YOUR_API_KEY";
const URL = "wss://speech-to-speech.us.assemblyai.com/v1/realtime";

const SAMPLE_RATE = 24000;

const ws = new WebSocket(URL, {
  headers: { Authorization: `Bearer ${API_KEY}` },
});

const speaker = new Speaker({
  channels: 1,
  bitDepth: 16,
  sampleRate: SAMPLE_RATE,
});

const microphone = mic({
  rate: SAMPLE_RATE,
  channels: 1,
  bitwidth: 16,
  encoding: "signed-integer",
});

const micStream = microphone.getAudioStream();
let sessionReady = false;

ws.on("open", () => {
  console.log("Connected");
  // Send config immediately on connect — before session.ready
  ws.send(JSON.stringify({
    type: "session.update",
    session: {
      system_prompt: "You are a helpful assistant.",
      tools: [],
    },
  }));
});

ws.on("message", (data) => {
  const event = JSON.parse(data.toString());

  switch (event.type) {
    case "session.ready":
      console.log(`Ready — start speaking (session_id=${event.session_id})`);
      // Gate mic start behind session.ready
      sessionReady = true;
      microphone.start();
      break;

    case "input.speech.started":
      console.log("Listening...");
      break;

    case "input.speech.stopped":
      console.log("Processing...");
      break;

    case "transcript.user":
      console.log(`You: ${event.text}`);
      break;

    case "reply.audio":
      speaker.write(Buffer.from(event.data, "base64"));
      break;

    case "transcript.agent":
      console.log(`Agent: ${event.text}`);
      break;

    case "session.error":
      console.error(`Error: ${event.message}`);
      ws.close();
      break;
  }
});

micStream.on("data", (chunk) => {
  // Only send audio after session.ready
  if (sessionReady && ws.readyState === WebSocket.OPEN) {
    ws.send(
      JSON.stringify({
        type: "input.audio",
        audio: chunk.toString("base64"),
      })
    );
  }
});
```

## Audio format

Both input and output audio use the same format:

| Property      | Value                                    |
| ------------- | ---------------------------------------- |
| Encoding      | PCM16 (16-bit signed integer, little-endian) |
| Sample rate   | 24,000 Hz                                |
| Channels      | Mono                                     |

Send audio as base64-encoded PCM16 chunks. Chunk size doesn't matter — the server buffers and processes continuously.

<Warning>
Do not send `input.audio` before the `session.ready` event fires. Sending audio before the session is established will produce a flood of `session.error` events with code `invalid_format`.
</Warning>

**Authentication**

## Events reference

### Client → Server

#### `input.audio`

Stream PCM16 audio to the agent. Send this continuously as the user speaks, but only after `session.ready` has been received.

```json
{
  "type": "input.audio",
  "audio": "<base64-encoded PCM16>"
}
```

| Field   | Type   | Description                       |
| ------- | ------ | --------------------------------- |
| `audio` | string | Base64-encoded PCM16 mono 24kHz audio |

---

#### `session.update`

Configure the session. Send this immediately on WebSocket connect — before `session.ready` arrives. Can also be sent multiple times mid-conversation to update the system prompt or tools.

```json
{
  "type": "session.update",
  "session": {
    "system_prompt": "You are a helpful assistant.",
    "greeting": "Hi! How can I help you today?",
    "tools": []
  }
}
```

| Field                    | Type   | Description                              |
| ------------------------ | ------ | ---------------------------------------- |
| `session.system_prompt`  | string | Sets the agent's personality and context |
| `session.greeting`       | string | What the agent says at the start of a conversation |
| `session.tools`          | array  | Tool definitions (see [Function calling](#function-calling)) |

---

#### `reply.create`

Manually trigger a response from the agent. Useful when VAD is disabled or you want to force a turn.

```json
{ "type": "reply.create" }
```

---

#### `reply.cancel`

Cancel the agent's current response (stop it from speaking). Use this to implement barge-in.

```json
{
  "type": "reply.cancel",
  "reply_id": "reply_abc123"
}
```

| Field      | Type   | Description                                  |
| ---------- | ------ | -------------------------------------------- |
| `reply_id` | string | The ID of the reply to cancel, from `reply.started` |

---

#### `session.resume`

Reconnect to an existing session using the `session_id` from a previous `session.ready` event. Useful for handling dropped connections without losing conversation context.

```json
{
  "type": "session.resume",
  "session_id": "sess_abc123"
}
```

| Field        | Type   | Description                                      |
| ------------ | ------ | ------------------------------------------------ |
| `session_id` | string | The session ID returned in the previous `session.ready` |

<Note>
If the session has expired, the server returns a `session.error` with code `session_not_found` or `session_forbidden`. In this case, start a fresh connection without `session.resume`.
</Note>

---

#### `tool.result`

Send the result of a tool call back to the agent.

```json
{
  "type": "tool.result",
  "call_id": "call_abc123",
  "result": "{\"temp_c\": \"22\", \"description\": \"Sunny\"}"
}
```

| Field     | Type   | Description                                |
| --------- | ------ | ------------------------------------------ |
| `call_id` | string | The `call_id` from the `tool.call` event   |
| `result`  | string | The result to return to the agent (JSON string recommended) |

---

### Server → Client

#### `session.ready`

Sent once the WebSocket connection is established and the session is ready to receive audio. Save `session_id` to enable reconnection via `session.resume`. Begin sending `input.audio` only after this event fires.

```json
{
  "type": "session.ready",
  "session_id": "sess_abc123"
}
```

| Field        | Type   | Description                                              |
| ------------ | ------ | -------------------------------------------------------- |
| `session_id` | string | Unique session identifier — use with `session.resume` to reconnect |

---

#### `session.updated`

Sent by the server after a `session.update` is successfully applied.

```json
{
  "type": "session.updated"
}
```

---

#### `input.speech.started`

Sent when the server detects the user has started speaking.

```json
{
  "type": "input.speech.started"
}
```

---

#### `input.speech.stopped`

Sent when the server detects the user has stopped speaking.

```json
{
  "type": "input.speech.stopped"
}
```

---

#### `transcript.user.delta`

Partial (interim) transcript of what the user is currently saying. Updates as the user speaks.

```json
{
  "type": "transcript.user.delta",
  "text": "What's the weather in"
}
```

---

#### `transcript.user`

Final transcript of the user's utterance, sent after `input.speech.stopped`.

```json
{
  "type": "transcript.user",
  "text": "What's the weather in Tokyo?",
  "item_id": "item_abc123"
}
```

---

#### `reply.started`

Sent when the agent begins generating a response.

```json
{
  "type": "reply.started",
  "reply_id": "reply_abc123"
}
```

| Field      | Type   | Description                                    |
| ---------- | ------ | ---------------------------------------------- |
| `reply_id` | string | Unique ID for this reply — use with `reply.cancel` |

---

#### `reply.audio`

A chunk of the agent's spoken response as base64 PCM16 audio. Arrives in real-time as the agent speaks — decode and play immediately for lowest latency.

```json
{
  "type": "reply.audio",
  "data": "<base64-encoded PCM16>"
}
```

---

#### `transcript.agent`

The full text of the agent's response, sent after the response completes.

```json
{
  "type": "transcript.agent",
  "text": "It's currently 22°C and sunny in Tokyo."
}
```

---

#### `reply.done`

Sent when the agent finishes speaking. If the agent was interrupted, the optional `status` field is set to `"interrupted"`.

```json
{
  "type": "reply.done"
}
```

```json
{
  "type": "reply.done",
  "status": "interrupted"
}
```

| Field    | Type   | Description                                              |
| -------- | ------ | -------------------------------------------------------- |
| `status` | string | Optional. Set to `"interrupted"` if the user barged in. |

---

#### `reply.interrupted`

Sent when the agent's response was interrupted (e.g. the user started speaking).

```json
{
  "type": "reply.interrupted"
}
```

---

#### `tool.call`

Sent when the agent wants to call a registered tool. The `arguments` field is a JSON string — parse it before use.

```json
{
  "type": "tool.call",
  "call_id": "call_abc123",
  "name": "get_weather",
  "arguments": "{\"location\": \"Tokyo\"}"
}
```

| Field       | Type   | Description                                        |
| ----------- | ------ | -------------------------------------------------- |
| `call_id`   | string | Unique ID — include this in `tool.result`          |
| `name`      | string | Name of the tool to call                           |
| `arguments` | string | JSON string of arguments — parse with `JSON.parse` |

---

#### `session.error`

Sent on connection or session errors.

```json
{
  "type": "session.error",
  "code": "session_not_found",
  "message": "Session not found or expired"
}
```

| Field     | Type   | Description |
| --------- | ------ | ----------- |
| `message` | string | Human-readable error description |
| `code`    | string | Machine-readable error code (see below) |

**Error codes:**

| Code | Description |
| ---- | ----------- |
| `session_not_found` | The `session_id` passed to `session.resume` does not exist |
| `session_forbidden` | The `session_id` belongs to a different API key |
| `invalid_format`    | The event sent by the client was malformed (e.g. audio sent before `session.ready`) |

## Session configuration

### System prompt

Set the agent's personality, role, and behaviour. Can be updated mid-session.

```json
{
  "type": "session.update",
  "session": {
    "system_prompt": "You are a friendly customer support agent for Acme Corp. Keep responses under 2 sentences. Never make up information — if you don't know, say so."
  }
}
```

**Tips for voice-first prompts:**

- Keep instructions concise — the model reads your prompt, not the user
- Use "Never say X" to ban specific phrases (e.g. "Never say 'Certainly' or 'Absolutely'")
- Instruct the agent to keep responses short: "Max 2 sentences per turn"
- For tool-based agents, tell the agent when and how to use each tool

### Greeting

Set what the agent says at the start of every conversation. This is different from the system prompt — it's spoken aloud to the user.

```json
{
  "type": "session.update",
  "session": {
    "system_prompt": "You are a helpful assistant.",
    "greeting": "Hi there! How can I help you today?"
  }
}
```

If no greeting is set, the agent waits silently for the user to speak first.

## Function calling

Register tools to let the agent take real-world actions. The agent decides when to call a tool based on the conversation.

### Register tools

Include tool definitions in `session.update`. Tools use an OpenAI-compatible format with a nested `"function"` key:

```json
{
  "type": "session.update",
  "session": {
    "system_prompt": "You are a helpful assistant. Use get_weather to look up weather when asked.",
    "tools": [
      {
        "type": "function",
        "function": {
          "name": "get_weather",
          "description": "Get the current weather for a city.",
          "parameters": {
            "type": "object",
            "properties": {
              "location": {
                "type": "string",
                "description": "City name, e.g. London"
              }
            },
            "required": ["location"]
          }
        }
      }
    ]
  }
}
```

### Handle tool calls

When the agent wants to call a tool, it emits a `tool.call` event. The `arguments` field is a JSON string — parse it before use. Execute the function and send back the result via `tool.result`:

<Tabs>

### Tool schema

Tools use an OpenAI-compatible format with a nested `"function"` key. Supported parameter types: `string`, `number`, `boolean`, `array`, `object`.

```json
{
  "type": "function",
  "function": {
    "name": "tool_name",
    "description": "What this tool does — the model uses this to decide when to call it",
    "parameters": {
      "type": "object",
      "properties": {
        "param": {
          "type": "string",
          "description": "What this parameter is"
        },
        "category": {
          "type": "string",
          "enum": ["option_a", "option_b", "option_c"]
        }
      },
      "required": ["param"]
    }
  }
}
```

## Interruptions

The server detects when the user starts speaking mid-response (`input.speech.started`) and will automatically cancel the current response. You can also cancel manually using the `reply_id` from `reply.started`:

```python
# Cancel the current agent reply
await ws.send(json.dumps({
    "type": "reply.cancel",
    "reply_id": current_reply_id,  # from reply.started
}))
```

Track the current `reply_id` by saving it when `reply.started` arrives, and clearing it on `reply.done` or `reply.interrupted`. If `reply.done` arrives with `status == "interrupted"`, the agent was barged in on.

## Browser integration

Browsers cannot set the `Authorization` header on WebSocket connections. The recommended pattern is a lightweight server-side proxy that adds the auth header before forwarding to AssemblyAI.

```
Browser ──── ws:// ──── Your proxy ──── wss://speech-to-speech.us.assemblyai.com
(no key)                 (adds auth)
```

**Minimal Node.js proxy:**

```javascript
import http from "http";
import { WebSocket, WebSocketServer } from "ws";

const AAI_URL = "wss://speech-to-speech.us.assemblyai.com/v1/realtime";
const API_KEY = process.env.ASSEMBLYAI_API_KEY;

const server = http.createServer();
const wss = new WebSocketServer({ server });

wss.on("connection", (clientWs) => {
  const aaiWs = new WebSocket(AAI_URL, {
    headers: { Authorization: `Bearer ${API_KEY}` },
  });

  clientWs.on("message", (data) => {
    if (aaiWs.readyState === WebSocket.OPEN) aaiWs.send(data);
  });

  aaiWs.on("message", (data) => {
    if (clientWs.readyState === WebSocket.OPEN) clientWs.send(data);
  });

  aaiWs.on("close", (code, reason) => clientWs.close(code, reason));
  clientWs.on("close", () => aaiWs.close());
});

server.listen(8080);
```

The browser then connects to `ws://localhost:8080` (or your deployed proxy URL) with no authentication needed.

## Event flow diagram

```
Client                              Server
  │                                   │
  │── WebSocket connect ──────────────►│
  │── session.update ─────────────────►│  (set system prompt + tools — sent immediately)
  │                                   │
  │◄─── session.ready ────────────────│  (save session_id for reconnection)
  │                                   │
  │── input.audio (stream) ──────────►│  (user speaks — only after session.ready)
  │── input.audio (stream) ──────────►│
  │── input.audio (stream) ──────────►│
  │                                   │
  │◄─── input.speech.started ─────────│  (VAD detected speech)
  │◄─── transcript.user.delta ────────│  (partial transcript)
  │◄─── transcript.user.delta ────────│
```
